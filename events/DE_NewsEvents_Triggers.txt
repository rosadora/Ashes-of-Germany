###########################
# News Events
###########################

add_namespace = DE_news_trigger

# Planes Crash In Nymmindegab (HIDDEN)
country_event = {
	id = DE_news_trigger.1

	trigger = {
		GER = {
			has_deployed_air_force_size = {
				    size > 1
				    type = tactical_bomber
				}
				has_war = yes
			}
	    DEN = { NOT = { has_war_with = GER } }
	}
	
	mean_time_to_happen = {
		days = 1
	}

	hidden = yes
	fire_only_once = yes

	immediate = {
		news_event = { id = DE_news.1 days = 36 }
	}

	option = { # A
		name = DE_news_trigger.1.a
	}
}
