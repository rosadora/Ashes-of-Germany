	desc = italia.1.d

			transfer_state = 156
			transfer_state = 115

			add_state_core = 849
			add_state_core = 156
			add_state_core = 115
			add_state_core = 114
			inherit_technology = ITA
			load_oob = ILF_civil_war
		}
		transfer_units_fraction= {
	     target = ILF
	     navy_ratio = 0.4
		 air_ratio = 0.3
		}
		declare_war_on = {
            target = ILF
            type = annex_everything
        }

		every_core_state = {
			add_core_of = NLC
		}
		NLC = {
			transfer_state = 158
			transfer_state = 159
			transfer_state = 160
			transfer_state = 39
			transfer_state = 850
			transfer_state = 736
			transfer_state = 852
			transfer_state = 161
			inherit_technology = ITA
			declare_war_on = {
            target = ILF
            type = annex_everything
            }
			load_oob = NLC_template
		}
		transfer_units_fraction= {
	     target = NLC
	     navy_ratio = 0.4
		 air_ratio = 0.3
		}
		declare_war_on = {
            target = NLC
            type = annex_everything
        }
		load_oob = ITA_fascist_side_civil_war
		mark_focus_tree_layout_dirty = yes
		remove_ideas = resistanza_moderata
		remove_ideas = azione_antifascista
		}
	}

	fire_only_once = yes

	option = {
		name = italia.1.a
		ai_chance = {
			base = 80
		}
	}
	option = {
		name = italia.1.b
		ai_chance = {
			base = 80
		}
		ILF = {
		change_tag_from = ITA
		}
	}
	option = {
		name = italia.1.c
		ai_chance = {
			base = 80
		}
		NLC = {
		change_tag_from = ITA
		}
	}
}
